## Getting Started

### 1. Set Up Development Environment

```bash
# Fork the repository on GitHub
# Clone your fork
git clone https://github.com/YOUR_USERNAME/testzeus-hercules.git
cd testzeus-hercules

# Add upstream remote
git remote add upstream https://github.com/test-zeus-ai/testzeus-hercules.git

# Create virtual environment
python -m venv venv
source venv/bin/activate  # On Windows: venv\Scripts\activate

# Install development dependencies
pip install -r requirements-dev.txt
pip install -e .

# Install pre-commit hooks
pre-commit install

# Install browser dependencies
playwright install
```

### 2. Verify Setup

```bash
# Run tests to ensure everything works
pytest tests/unit/

# Run linting and formatting
pre-commit run --all-files

# Test basic functionality
echo 'Feature: Test
Scenario: Simple test
  Given I have a web browser open
  When I navigate to "https://example.com"
  Then I should see "Example Domain"' > test.feature

hercules test.feature
```

### 2. Make Changes

Follow our coding standards:

```python
# Example: Adding a new tool
from testzeus_hercules.core.tools.base_tool import BaseTool

class CustomValidationTool(BaseTool):
    name = "custom_validation_tool"
    description = "Performs custom business logic validation"
    
    def __init__(self, **kwargs):
        super().__init__(**kwargs)
        self.validation_rules = kwargs.get('validation_rules', {})
    
    def execute(self, data: dict, rules: list = None) -> dict:
        rules = rules or self.validation_rules
        results = {
            "valid": True,
            "errors": [],
            "warnings": []
        }
        
        for rule in rules:
            validation_result = self.apply_rule(data, rule)
            if not validation_result.valid:
                results["valid"] = False
                results["errors"].extend(validation_result.errors)
        
        return results
```

### 3. Write Tests

```python
# tests/unit/tools/test_custom_validation_tool.py
import pytest
from testzeus_hercules.core.tools.custom_validation_tool import CustomValidationTool

class TestCustomValidationTool:
    def test_valid_data(self):
        tool = CustomValidationTool()
        result = tool.execute({"email": "test@example.com"})
        assert result["valid"] is True
        assert len(result["errors"]) == 0
    
    def test_invalid_data(self):
        tool = CustomValidationTool()
        result = tool.execute({"email": "invalid-email"})
        assert result["valid"] is False
        assert len(result["errors"]) > 0
```

### Error Handling

```python
# Good: Specific exception handling
try:
    result = self.execute_action(action, parameters)
except TimeoutError as e:
    logger.error(f"Action timed out: {e}")
    return {"success": False, "error": "timeout"}
except ValidationError as e:
    logger.error(f"Validation failed: {e}")
    return {"success": False, "error": "validation"}
except Exception as e:
    logger.error(f"Unexpected error: {e}")
    return {"success": False, "error": "unknown"}
```

## Testing Guidelines

### Unit Tests

```python
# Good: Focused, isolated tests
class TestAPINavigationAgent:
    @pytest.fixture
    def agent(self):
        return APINavigationAgent(base_url="https://api.example.com")
    
    def test_get_request_success(self, agent, mock_requests):
        mock_requests.get.return_value.status_code = 200
        mock_requests.get.return_value.json.return_value = {"id": 1}
        
        result = agent.execute_get_request("/users/1")
        
        assert result["success"] is True
        assert result["data"]["id"] == 1
        mock_requests.get.assert_called_once_with(
            "https://api.example.com/users/1",
            headers=agent.default_headers
        )
```

### Integration Tests

```python
# Good: End-to-end workflow testing
class TestUserRegistrationFlow:
    def test_complete_registration_flow(self, test_app):
        # Test the complete user registration process
        registration_data = {
            "email": "test@example.com",
            "password": "SecurePass123!",
            "first_name": "Test",
            "last_name": "User"
        }
        
        # Register user
        response = test_app.post("/api/register", json=registration_data)
        assert response.status_code == 201
        
        # Verify user can login
        login_response = test_app.post("/api/login", json={
            "email": registration_data["email"],
            "password": registration_data["password"]
        })
        assert login_response.status_code == 200
        assert "token" in login_response.json()
```

## Advanced Contributing

### Custom Agent Development

```python
from testzeus_hercules.core.agents.base_nav_agent import BaseNavAgent

class CustomIntegrationAgent(BaseNavAgent):
    agent_name = "custom_integration_agent"
    
    def __init__(self, **kwargs):
        super().__init__(**kwargs)
        self.integration_config = kwargs.get('integration_config', {})
    
    def execute_action(self, action: str, parameters: dict) -> dict:
        if action == "sync_data":
            return self.sync_external_data(parameters)
        elif action == "validate_integration":
            return self.validate_integration_status(parameters)
        else:
            return {"error": f"Unknown action: {action}"}
```

### Custom Tool Development

```python
from testzeus_hercules.core.tools.base_tool import BaseTool

class CustomReportingTool(BaseTool):
    name = "custom_reporting_tool"
    description = "Generates custom test reports"
    
    def execute(self, test_results: list, format: str = "html") -> dict:
        if format == "html":
            return self.generate_html_report(test_results)
        elif format == "pdf":
            return self.generate_pdf_report(test_results)
        else:
            return {"error": f"Unsupported format: {format}"}
```
